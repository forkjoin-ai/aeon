% arXiv preamble: fix monospace overflow and table wrapping
\usepackage[margin=1in]{geometry}

% Allow line breaks in \texttt (monospace) at / . - _
\renewcommand{\texttt}[1]{%
  \begingroup
  \ttfamily
  \hyphenchar\font=`\-
  \chardef\breakchar=`\/
  \lccode`~=\breakchar
  \lowercase{\let~}\allowbreak
  #1%
  \endgroup
}

% Shrink monospace font slightly to reduce overflow
\makeatletter
\DeclareFontFamily{T1}{lmtt}{}
\DeclareFontShape{T1}{lmtt}{m}{n}{<->s*[0.85]ec-lmtt10}{}
\makeatother
